019    package org.apache.hadoop.mapred.lib.aggregate;
020    
021    import org.apache.hadoop.classification.InterfaceAudience;
022    import org.apache.hadoop.classification.InterfaceStability;
023    
024    /**
025     * This class implements a value aggregator that dedupes a sequence of objects.
026     */
027    @InterfaceAudience.Public
028    @InterfaceStability.Stable
029    public class UniqValueCount 
030        extends org.apache.hadoop.mapreduce.lib.aggregate.UniqValueCount 
031        implements ValueAggregator<Object> {
032      /**
033       * the default constructor
034       * 
035       */
036      public UniqValueCount() {
037        super();
038      }
039      
040      /**
041       * constructor
042       * @param maxNum the limit in the number of unique values to keep.
043       *  
044       */
045      public UniqValueCount(long maxNum) {
046        super(maxNum);
047      }
048    }
