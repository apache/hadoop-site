018    package org.apache.hadoop.mapreduce.task.annotation;
019    
020    import java.lang.annotation.Documented;
021    import java.lang.annotation.ElementType;
022    import java.lang.annotation.Retention;
023    import java.lang.annotation.RetentionPolicy;
024    import java.lang.annotation.Target;
025    
026    import org.apache.hadoop.classification.InterfaceAudience;
027    import org.apache.hadoop.classification.InterfaceStability;
028    
029    /**
030     * Contract representing to the framework that the task can be safely preempted
031     * and restarted between invocations of the user-defined function.
032     *
033     * This is often true when the result of a function does not rely on state
034     * derived from previous elements in the record stream, but the guarantee is
035     * left as an exercise to the implementor.
036     */
037    @Documented
038    @Target(ElementType.TYPE)
039    @Retention(RetentionPolicy.RUNTIME)
040    @InterfaceAudience.Public
041    @InterfaceStability.Evolving
042    public @interface Checkpointable { }
