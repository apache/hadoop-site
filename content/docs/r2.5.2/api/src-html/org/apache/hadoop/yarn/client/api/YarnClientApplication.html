019    package org.apache.hadoop.yarn.client.api;
020    
021    import org.apache.hadoop.classification.InterfaceAudience;
022    import org.apache.hadoop.classification.InterfaceStability;
023    import org.apache.hadoop.yarn.api.protocolrecords.GetNewApplicationResponse;
024    import org.apache.hadoop.yarn.api.records.ApplicationSubmissionContext;
025    
026    @InterfaceAudience.Public
027    @InterfaceStability.Stable
028    
029    /**
030     * Holder for the {@link GetNewApplicationResponse} and {@link
031     * ApplicationSubmissionContext} objects created via {@link org.apache.hadoop
032     * .yarn.client.api.YarnClient#createApplication()}
033     */
034    public  class YarnClientApplication {
035      private final GetNewApplicationResponse newAppResponse;
036      private final ApplicationSubmissionContext appSubmissionContext;
037    
038      public YarnClientApplication(GetNewApplicationResponse newAppResponse,
039                                   ApplicationSubmissionContext appContext) {
040        this.newAppResponse = newAppResponse;
041        this.appSubmissionContext = appContext;
042      }
043    
044      public GetNewApplicationResponse getNewApplicationResponse() {
045        return newAppResponse;
046      }
047    
048      public ApplicationSubmissionContext getApplicationSubmissionContext() {
049        return appSubmissionContext;
050      }
051    }
