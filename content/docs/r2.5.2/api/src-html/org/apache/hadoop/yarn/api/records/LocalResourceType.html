019    package org.apache.hadoop.yarn.api.records;
020    
021    import org.apache.hadoop.classification.InterfaceAudience.Public;
022    import org.apache.hadoop.classification.InterfaceStability.Stable;
023    import org.apache.hadoop.yarn.api.ContainerManagementProtocol;
024    
025    /**
026     * <p><code>LocalResourceType</code> specifies the <em>type</em>
027     * of a resource localized by the <code>NodeManager</code>.</p> 
028     *
029     * <p>The <em>type</em> can be one of:
030     *   <ul>
031     *     <li>
032     *       {@link #FILE} - Regular file i.e. uninterpreted bytes. 
033     *     </li>
034     *     <li>
035     *       {@link #ARCHIVE} - Archive, which is automatically unarchived by the 
036     *       <code>NodeManager</code>.
037     *     </li>
038     *     <li>
039     *       {@link #PATTERN} - A hybrid between {@link #ARCHIVE} and {@link #FILE}.
040     *   </ul>
041     * </p>
042     *
043     * @see LocalResource
044     * @see ContainerLaunchContext
045     * @see ApplicationSubmissionContext
046     * @see ContainerManagementProtocol#startContainers(org.apache.hadoop.yarn.api.protocolrecords.StartContainersRequest)
047     */
048    @Public
049    @Stable
050    public enum LocalResourceType {
051      
052      /**
053       * Archive, which is automatically unarchived by the <code>NodeManager</code>.
054       */
055      ARCHIVE,
056      
057      /**
058       * Regular file i.e. uninterpreted bytes.
059       */
060      FILE,
061      
062      /**
063       * A hybrid between archive and file.  Only part of the file is unarchived,
064       * and the original file is left in place, but in the same directory as the
065       * unarchived part.  The part that is unarchived is determined by pattern
066       * in #{@link LocalResource}.  Currently only jars support pattern, all
067       * others will be treated like a #{@link LocalResourceType#ARCHIVE}.
068       */
069      PATTERN
070    }
