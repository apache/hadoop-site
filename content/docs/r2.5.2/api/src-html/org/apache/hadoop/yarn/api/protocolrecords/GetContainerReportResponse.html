019    package org.apache.hadoop.yarn.api.protocolrecords;
020    
021    import org.apache.hadoop.classification.InterfaceAudience.Public;
022    import org.apache.hadoop.classification.InterfaceStability.Unstable;
023    import org.apache.hadoop.yarn.api.records.ContainerReport;
024    import org.apache.hadoop.yarn.util.Records;
025    
026    /**
027     * <p>
028     * The response sent by the <code>ResourceManager</code> to a client requesting
029     * a container report.
030     * </p>
031     * 
032     * <p>
033     * The response includes a {@link ContainerReport} which has details of a
034     * container.
035     * </p>
036     * 
037     */
038    @Public
039    @Unstable
040    public abstract class GetContainerReportResponse {
041      @Public
042      @Unstable
043      public static GetContainerReportResponse newInstance(
044          ContainerReport containerReport) {
045        GetContainerReportResponse response =
046            Records.newRecord(GetContainerReportResponse.class);
047        response.setContainerReport(containerReport);
048        return response;
049      }
050    
051      /**
052       * Get the <code>ContainerReport</code> for the container.
053       * 
054       * @return <code>ContainerReport</code> for the container
055       */
056      @Public
057      @Unstable
058      public abstract ContainerReport getContainerReport();
059    
060      @Public
061      @Unstable
062      public abstract void setContainerReport(ContainerReport containerReport);
063    }
