019    package org.apache.hadoop.mapred.lib;
020    
021    import org.apache.hadoop.classification.InterfaceAudience;
022    import org.apache.hadoop.classification.InterfaceStability;
023    import org.apache.hadoop.mapred.JobConf;
024    import org.apache.hadoop.mapred.JobConfigurable;
025    import org.apache.hadoop.mapreduce.JobContext;
026    
027    /**
028     * This comparator implementation provides a subset of the features provided
029     * by the Unix/GNU Sort. In particular, the supported features are:
030     * -n, (Sort numerically)
031     * -r, (Reverse the result of comparison)
032     * -k pos1[,pos2], where pos is of the form f[.c][opts], where f is the number
033     *  of the field to use, and c is the number of the first character from the
034     *  beginning of the field. Fields and character posns are numbered starting
035     *  with 1; a character position of zero in pos2 indicates the field's last
036     *  character. If '.c' is omitted from pos1, it defaults to 1 (the beginning
037     *  of the field); if omitted from pos2, it defaults to 0 (the end of the
038     *  field). opts are ordering options (any of 'nr' as described above). 
039     * We assume that the fields in the key are separated by
040     * {@link JobContext#MAP_OUTPUT_KEY_FIELD_SEPERATOR} 
041     */
042    @InterfaceAudience.Public
043    @InterfaceStability.Stable
044    public class KeyFieldBasedComparator<K, V> extends 
045        org.apache.hadoop.mapreduce.lib.partition.KeyFieldBasedComparator<K, V>
046        implements JobConfigurable {
047    
048      public void configure(JobConf job) {
049        super.setConf(job);
050      }
051    }
