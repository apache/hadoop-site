019    package org.apache.hadoop.yarn.api.protocolrecords;
020    
021    import org.apache.hadoop.classification.InterfaceAudience.Private;
022    import org.apache.hadoop.classification.InterfaceAudience.Public;
023    import org.apache.hadoop.classification.InterfaceStability.Stable;
024    import org.apache.hadoop.classification.InterfaceStability.Unstable;
025    import org.apache.hadoop.yarn.api.ApplicationClientProtocol;
026    import org.apache.hadoop.yarn.api.records.ApplicationId;
027    import org.apache.hadoop.yarn.api.records.Resource;
028    import org.apache.hadoop.yarn.util.Records;
029    
030    /**
031     * <p>The response sent by the <code>ResourceManager</code> to the client for 
032     * a request to get a new {@link ApplicationId} for submitting applications.</p>
033     * 
034     * <p>Clients can submit an application with the returned
035     * {@link ApplicationId}.</p>
036     *
037     * @see ApplicationClientProtocol#getNewApplication(GetNewApplicationRequest)
038     */
039    @Public
040    @Stable
041    public abstract class GetNewApplicationResponse {
042    
043      @Private
044      @Unstable
045      public static GetNewApplicationResponse newInstance(
046          ApplicationId applicationId, Resource minCapability,
047          Resource maxCapability) {
048        GetNewApplicationResponse response =
049            Records.newRecord(GetNewApplicationResponse.class);
050        response.setApplicationId(applicationId);
051        response.setMaximumResourceCapability(maxCapability);
052        return response;
053      }
054    
055      /**
056       * Get the <em>new</em> <code>ApplicationId</code> allocated by the 
057       * <code>ResourceManager</code>.
058       * @return <em>new</em> <code>ApplicationId</code> allocated by the 
059       *          <code>ResourceManager</code>
060       */
061      @Public
062      @Stable
063      public abstract ApplicationId getApplicationId();
064    
065      @Private
066      @Unstable
067      public abstract void setApplicationId(ApplicationId applicationId);
068    
069      /**
070       * Get the maximum capability for any {@link Resource} allocated by the 
071       * <code>ResourceManager</code> in the cluster.
072       * @return maximum capability of allocated resources in the cluster
073       */
074      @Public
075      @Stable
076      public abstract Resource getMaximumResourceCapability();
077      
078      @Private
079      @Unstable
080      public abstract void setMaximumResourceCapability(Resource capability); 
081    }
