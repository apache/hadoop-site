018    package org.apache.hadoop.fs;
019    
020    import java.io.*;
021    
022    import org.apache.hadoop.classification.InterfaceAudience;
023    import org.apache.hadoop.classification.InterfaceStability;
024    
025    /**
026     *  Stream that permits seeking.
027     */
028    @InterfaceAudience.Public
029    @InterfaceStability.Evolving
030    public interface Seekable {
031      /**
032       * Seek to the given offset from the start of the file.
033       * The next read() will be from that location.  Can't
034       * seek past the end of the file.
035       */
036      void seek(long pos) throws IOException;
037      
038      /**
039       * Return the current offset from the start of the file
040       */
041      long getPos() throws IOException;
042    
043      /**
044       * Seeks a different copy of the data.  Returns true if 
045       * found a new source, false otherwise.
046       */
047      @InterfaceAudience.Private
048      boolean seekToNewSource(long targetPos) throws IOException;
049    }
