019    package org.apache.hadoop.mapreduce.lib.map;
020    
021    import java.io.IOException;
022    import java.util.StringTokenizer;
023    
024    import org.apache.hadoop.classification.InterfaceAudience;
025    import org.apache.hadoop.classification.InterfaceStability;
026    import org.apache.hadoop.io.IntWritable;
027    import org.apache.hadoop.io.Text;
028    import org.apache.hadoop.mapreduce.Mapper;
029    
030    /**
031     * Tokenize the input values and emit each word with a count of 1.
032     */
033    @InterfaceAudience.Public
034    @InterfaceStability.Stable
035    public class TokenCounterMapper extends Mapper<Object, Text, Text, IntWritable>{
036        
037      private final static IntWritable one = new IntWritable(1);
038      private Text word = new Text();
039      
040      @Override
041      public void map(Object key, Text value, Context context
042                      ) throws IOException, InterruptedException {
043        StringTokenizer itr = new StringTokenizer(value.toString());
044        while (itr.hasMoreTokens()) {
045          word.set(itr.nextToken());
046          context.write(word, one);
047        }
048      }
049    }
