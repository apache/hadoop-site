018    package org.apache.hadoop.security;
019    
020    import java.io.IOException;
021    import java.util.List;
022    
023    import org.apache.hadoop.classification.InterfaceAudience;
024    import org.apache.hadoop.classification.InterfaceStability;
025    
026    /**
027     * An interface for the implementation of a user-to-groups mapping service
028     * used by {@link Groups}.
029     */
030    @InterfaceAudience.Public
031    @InterfaceStability.Evolving
032    public interface GroupMappingServiceProvider {
033      
034      /**
035       * Get all various group memberships of a given user.
036       * Returns EMPTY list in case of non-existing user
037       * @param user User's name
038       * @return group memberships of user
039       * @throws IOException
040       */
041      public List<String> getGroups(String user) throws IOException;
042      /**
043       * Refresh the cache of groups and user mapping
044       * @throws IOException
045       */
046      public void cacheGroupsRefresh() throws IOException;
047      /**
048       * Caches the group user information
049       * @param groups list of groups to add to cache
050       * @throws IOException
051       */
052      public void cacheGroupsAdd(List<String> groups) throws IOException;
053    }
