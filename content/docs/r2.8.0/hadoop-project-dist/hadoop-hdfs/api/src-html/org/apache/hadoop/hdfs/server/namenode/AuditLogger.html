018package org.apache.hadoop.hdfs.server.namenode;
019
020import java.net.InetAddress;
021import java.security.Principal;
022
023import org.apache.hadoop.classification.InterfaceAudience;
024import org.apache.hadoop.classification.InterfaceStability;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.FileStatus;
027
028/**
029 * Interface defining an audit logger.
030 */
031@InterfaceAudience.Public
032@InterfaceStability.Evolving
033public interface AuditLogger {
034
035  /**
036   * Called during initialization of the logger.
037   *
038   * @param conf The configuration object.
039   */
040  void initialize(Configuration conf);
041
042  /**
043   * Called to log an audit event.
044   * <p>
045   * This method must return as quickly as possible, since it's called
046   * in a critical section of the NameNode's operation.
047   *
048   * @param succeeded Whether authorization succeeded.
049   * @param userName Name of the user executing the request.
050   * @param addr Remote address of the request.
051   * @param cmd The requested command.
052   * @param src Path of affected source file.
053   * @param dst Path of affected destination file (if any).
054   * @param stat File information for operations that change the file's
055   *             metadata (permissions, owner, times, etc).
056   */
057  void logAuditEvent(boolean succeeded, String userName,
058      InetAddress addr, String cmd, String src, String dst,
059      FileStatus stat);
060
061}
