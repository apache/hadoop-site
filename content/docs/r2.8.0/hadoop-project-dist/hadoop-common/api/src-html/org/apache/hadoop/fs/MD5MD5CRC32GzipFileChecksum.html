019package org.apache.hadoop.fs;
020
021import org.apache.hadoop.io.MD5Hash;
022import org.apache.hadoop.util.DataChecksum;
023
024/** For CRC32 with the Gzip polynomial */
025public class MD5MD5CRC32GzipFileChecksum extends MD5MD5CRC32FileChecksum {
026  /** Same as this(0, 0, null) */
027  public MD5MD5CRC32GzipFileChecksum() {
028    this(0, 0, null);
029  }
030
031  /** Create a MD5FileChecksum */
032  public MD5MD5CRC32GzipFileChecksum(int bytesPerCRC, long crcPerBlock, MD5Hash md5) {
033    super(bytesPerCRC, crcPerBlock, md5);
034  }
035  @Override
036  public DataChecksum.Type getCrcType() {
037    // default to the one that is understood by all releases.
038    return DataChecksum.Type.CRC32;
039  }
040}
