019    package org.apache.hadoop.lib.wsrs;
020    
021    import org.apache.hadoop.classification.InterfaceAudience;
022    import org.apache.hadoop.io.IOUtils;
023    
024    import javax.ws.rs.core.StreamingOutput;
025    import java.io.IOException;
026    import java.io.InputStream;
027    import java.io.OutputStream;
028    
029    @InterfaceAudience.Private
030    public class InputStreamEntity implements StreamingOutput {
031      private InputStream is;
032      private long offset;
033      private long len;
034    
035      public InputStreamEntity(InputStream is, long offset, long len) {
036        this.is = is;
037        this.offset = offset;
038        this.len = len;
039      }
040    
041      public InputStreamEntity(InputStream is) {
042        this(is, 0, -1);
043      }
044    
045      @Override
046      public void write(OutputStream os) throws IOException {
047        IOUtils.skipFully(is, offset);
048        if (len == -1) {
049          IOUtils.copyBytes(is, os, 4096, true);
050        } else {
051          IOUtils.copyBytes(is, os, len, true);
052        }
053      }
054    }
