019    package org.apache.hadoop.yarn.api.protocolrecords;
020    
021    import org.apache.hadoop.classification.InterfaceAudience.Private;
022    import org.apache.hadoop.classification.InterfaceAudience.Public;
023    import org.apache.hadoop.classification.InterfaceStability.Stable;
024    import org.apache.hadoop.classification.InterfaceStability.Unstable;
025    import org.apache.hadoop.yarn.api.ApplicationClientProtocol;
026    import org.apache.hadoop.yarn.api.records.YarnClusterMetrics;
027    import org.apache.hadoop.yarn.util.Records;
028    
029    /**
030     * <p>The response sent by the <code>ResourceManager</code> to a client
031     * requesting cluster metrics.<p>
032     * 
033     * @see YarnClusterMetrics
034     * @see ApplicationClientProtocol#getClusterMetrics(GetClusterMetricsRequest)
035     */
036    @Public
037    @Stable
038    public abstract class GetClusterMetricsResponse {
039    
040      @Private
041      @Unstable
042      public static GetClusterMetricsResponse
043          newInstance(YarnClusterMetrics metrics) {
044        GetClusterMetricsResponse response =
045            Records.newRecord(GetClusterMetricsResponse.class);
046        response.setClusterMetrics(metrics);
047        return response;
048      }
049    
050      /**
051       * Get the <code>YarnClusterMetrics</code> for the cluster.
052       * @return <code>YarnClusterMetrics</code> for the cluster
053       */
054      @Public
055      @Stable
056      public abstract YarnClusterMetrics getClusterMetrics();
057      
058      @Private
059      @Unstable
060      public abstract void setClusterMetrics(YarnClusterMetrics metrics);
061    }
