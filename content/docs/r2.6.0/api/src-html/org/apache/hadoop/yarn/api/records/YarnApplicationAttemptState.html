019    package org.apache.hadoop.yarn.api.records;
020    
021    import org.apache.hadoop.classification.InterfaceAudience.Public;
022    import org.apache.hadoop.classification.InterfaceStability.Stable;
023    
024    /**
025     * Enumeration of various states of a <code>RMAppAttempt</code>.
026     */
027    @Public
028    @Stable
029    public enum YarnApplicationAttemptState {
030      /** AppAttempt was just created. */
031      NEW,
032    
033      /** AppAttempt has been submitted. */
034      SUBMITTED,
035    
036      /** AppAttempt was scheduled */
037      SCHEDULED,
038    
039      /** Acquired AM Container from Scheduler and Saving AppAttempt Data */
040      ALLOCATED_SAVING,
041    
042      /** AppAttempt Data was saved */
043      ALLOCATED,
044    
045      /** AppAttempt was launched */
046      LAUNCHED,
047    
048      /** AppAttempt failed. */
049      FAILED,
050    
051      /** AppAttempt is currently running. */
052      RUNNING,
053    
054      /** AppAttempt is finishing. */
055      FINISHING,
056    
057      /** AppAttempt finished successfully. */
058      FINISHED,
059    
060      /** AppAttempt was terminated by a user or admin. */
061      KILLED
062    
063    }
